* Defuns
**** Reload
Reloads the configuration
#+BEGIN_SRC emacs-lisp
(defun alex/reload ()
  (interactive)
  (load-file (expand-file-name "init.el" user-emacs-directory)))
(provide 'alex/reload)
#+END_SRC
